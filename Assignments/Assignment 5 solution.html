2. SELECT name FROM courses WHERE instructor IN (SELECT DISTINCT instructor FROM courses WHERE gender =

3. SELECT name, grade FROM enrollments JOIN students ON
enrollments.student_id = students.student_id WHERE students.student_id = 2;
4. SELECT AVG(age) FROM students WHERE gender = 'Male';
5. SELECT name, age FROM students ORDER BY age ASC LIMIT 3;
6. SELECT students.name FROM students JOIN enrollments ON
students.student_id = enrollments.student_id GROUP BY students.student_id HAVING
COUNT(DISTINCT enrollments.course_id) > 2;
7. SELECT courses.name, AVG(grade) FROM courses JOIN enrollments ON
courses.course_id = enrollments.course_id GROUP BY courses.course_id ORDER BY AVG(grade) DESC;
8. SELECT students.name, courses.name FROM students JOIN enrollments ON students.student_id =
enrollments.student_id JOIN courses ON
enrollments.course_id = courses.course_id ORDER BY students.name ASC;
9. SELECT gender, COUNT(*) FROM students GROUP BY gender;
